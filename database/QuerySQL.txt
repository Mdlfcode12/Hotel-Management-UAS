-- 1. Tabel User (AdminResepsionis)
CREATE TABLE tabel_user (
    id_user INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    password VARCHAR(50) NOT NULL,
    nama_lengkap VARCHAR(100)
);

-- 2. Tabel Kamar
CREATE TABLE tabel_kamar (
    kode_kamar VARCHAR(10) PRIMARY KEY,
    tipe_kamar VARCHAR(50),
    harga INT,
    status VARCHAR(20) -- Isi 'Tersedia' atau 'Terisi'
);

-- 3. Tabel Tamu
CREATE TABLE tabel_tamu (
    nik VARCHAR(20) PRIMARY KEY,
    nama_tamu VARCHAR(100),
    no_telp VARCHAR(15),
    alamat TEXT
);

-- 4. Tabel Reservasi (Transaksi)
CREATE TABLE tabel_reservasi (
    id_reservasi VARCHAR(20) PRIMARY KEY,
    tgl_checkin DATE,
    tgl_checkout DATE,
    nik_tamu VARCHAR(20),
    kode_kamar VARCHAR(10),
    total_bayar INT,
    id_user INT,
    FOREIGN KEY (nik_tamu) REFERENCES tabel_tamu(nik),
    FOREIGN KEY (kode_kamar) REFERENCES tabel_kamar(kode_kamar),
    FOREIGN KEY (id_user) REFERENCES tabel_user(id_user)
);
